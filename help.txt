#include <stdlib.h>  // обязательно для malloc и free

void* malloc(size_t size);  // выделяет size байт памяти
void free(void* ptr);       // освобождает память

* - означает, что это указатель.

malloc(Количество_элементов × sizeof(Тип))

Примеры:
malloc(10 * sizeof(char))
10 × 1 = 10 байт
malloc(10 * sizeof(int))
10 × 4 = 40 байт
malloc(10 * sizeof(double))
10 × 8 = 80 байт
malloc(5 * sizeof(long long))
5 × 8 = 40 байт
